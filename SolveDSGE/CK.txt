# CK 2 shock model from "Indirect Likelihood Inference"

states:
k, z, η
end

jumps:
y, c, n, r, w, MUC, MUL
end

shocks:
ϵ
u
end

parameters:
α = 0.33
β = 0.99
δ = 0.025
γ = 2.0
ρ₁ = 0.9
σ₁ = 0.02 
ρ₂ = 0.7
σ₂ = 0.01
ψ =  3.41744
end

equations:
MUC = c^(-γ)
MUL = ψ*exp(η)
r = α * exp(z) * k^(α-1) * n^(1-α)
w = (1-α)*exp(z)* k^α * n^(-α)
MUC = β*MUC(+1) * (1 + r(+1) - δ)
MUL/MUC = w
z(+1) = ρ₁*z + σ₁*u
η(+1) = ρ₂*η + σ₂*ϵ
y = exp(z) * (k^α) * (n^(1-α))
k(+1) = y - c + (1-δ)*k
end
