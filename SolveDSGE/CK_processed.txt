function nlsolve_static_equations(f::Array{T,1},x::Array{T,1},param::Array{T1,1}) where {T<:Number, T1<:Real} 
 
  f[1] = x[1] - (param[5]*x[1] + param[6] * 0.0)
  f[2] = x[2] - (param[7]*x[2] + param[8] * 0.0)
  f[3] = x[7] - (param[1] * exp(x[1]) * x[3]^(param[1]-1) * x[6]^(1-param[1]))
  f[4] = x[8] - ((1-param[1])*exp(x[1])* x[3]^param[1] * x[6]^(-param[1]))
  f[5] = x[9] - (param[2]*x[9] * (1 + x[7] - param[3]))
  f[6] = x[8] - (x[10]/x[9])
  f[7] = x[9] - (x[5]^(-param[4]))
  f[8] = x[10] - (param[9]*exp(x[2]))
  f[9] = x[4] - (exp(x[1]) * (x[3]^param[1]) * (x[6]^(1-param[1])))
  f[10] = x[3] - (x[4] - x[5] + (1-param[3])*x[3])

end

function closure_projection_equations(state,scaled_weights,order,domain,approximate,param) 
 
  function projection_equations(f::Array{T,1},x::Array{T,1}) where {T<:Number} 
 
    approx4 = approximate(scaled_weights[1],x[7+1:end],order,domain)
    approx6 = approximate(scaled_weights[2],x[7+1:end],order,domain)

    #f = Array{T}(undef,10) 
 
    f[1] = x[8] - (param[5]*state[1] + param[6] * 0.0)
    f[2] = x[9] - (param[7]*state[2] + param[8] * 0.0)
    f[3] = x[4] - (param[1] * exp(state[1]) * state[3]^(param[1]-1) * x[3]^(1-param[1]))
    f[4] = x[5] - ((1-param[1])*exp(state[1])* state[3]^param[1] * x[3]^(-param[1]))
    f[5] = x[6] - (param[2]*approx6 * (1 + approx4 - param[3]))
    f[6] = x[5] - (x[7]/x[6])
    f[7] = x[6] - (x[2]^(-param[4]))
    f[8] = x[7] - (param[9]*exp(state[2]))
    f[9] = x[1] - (exp(state[1]) * (state[3]^param[1]) * (x[3]^(1-param[1])))
    f[10] = x[10] - (x[1] - x[2] + (1-param[3])*state[3])

    #return f 
 
  end 
 
  return projection_equations 
 
end
